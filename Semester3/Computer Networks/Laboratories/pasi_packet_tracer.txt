		=> router x -> cli ->cod

	DYNAMIC NAT -> translating all trafic from the lan

	=> router x -> cli ->cod

enable
configure terminal
int f0/0 
ip nat outside // if f0/0 is towards outside
int f0/1
ip nat inside
exit
acces-list 1 permit _LAN_IP_ _LAN_MASK_
ip nat pool POOL1 100.0.0.1 100.0.0.2 netmask 255.255.255.0 //pool will have only 2 translating addresses
ip nat inside source list 1 pool POOL1
